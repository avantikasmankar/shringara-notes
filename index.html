    <script>

        let notes = JSON.parse(localStorage.getItem('navrasaNotes') || '[]');

        

        function showTab(tabName) {

            // Hide all tabs

            document.querySelectorAll('.tab-content').forEach(content => {

                content.classList.remove('active');

            });

            document.querySelectorAll('.tab').forEach(tab => {

                tab.classList.remove('active');

            });

            

            // Show selected tab

            document.getElementById(tabName + '-tab').classList.add('active');

            event.target.classList.add('active');

            

            if (tabName === 'list') {

                loadNotes();

            }

        }

        

        function addNote() {

            const input = document.getElementById('noteInput');

            const rasaSelector = document.getElementById('rasaSelector');

            const text = input.value.trim();

            const selectedRasa = rasaSelector.value;

            

            if (text === '') {

                alert('Please type something before adding!');

                return;

            }

            

            if (selectedRasa === '') {

                alert('Please select an emotion for your note!');

                return;

            }

            

            const newNote = {

                text: text,

                rasa: selectedRasa,

                date: new Date().toLocaleDateString(),

                time: new Date().toLocaleTimeString(),

                id: Date.now()

            };

            

            notes.push(newNote);

            saveNotes();

            

            // Clear inputs

            input.value = '';

            rasaSelector.value = '';

            

            // Show success message

            const successMsg = document.getElementById('successMessage');

            successMsg.style.display = 'block';

            setTimeout(() => {

                successMsg.style.display = 'none';

            }, 2000);

            

            // Update counter

            updateCounter();

        }

        

        function loadNotes() {

            const container = document.getElementById('notesContainer');

            

            if (notes.length === 0) {

                container.innerHTML = `

                    <div class="empty-state">

                        No notes yet! Go to "Add Note" tab to create your first note.

                    </div>

                `;

                return;

            }

            

            container.innerHTML = '';

            // Show newest notes first

            [...notes].reverse().forEach((note) => {

                const rasaTag = `<div class="rasa-tag rasa-${note.rasa}">${getRasaDisplay(note.rasa)}</div>`;

                

                const noteElement = document.createElement('div');

                noteElement.className = 'note-item';

                noteElement.innerHTML = `

                    <button onclick="deleteNote(${note.id})" class="delete-btn">×</button>

                    ${rasaTag}

                    <div class="note-text">${escapeHtml(note.text)}</div>

                    <div class="note-meta">Added on ${note.date} at ${note.time}</div>

                `;

                container.appendChild(noteElement);

            });

        }

        

        function getRasaDisplay(rasa) {

            const rasaNames = {

                'shringara': 'Shringara (Love)',

                'hasya': 'Hasya (Laughter)',

                'karuna': 'Karuna (Compassion)',

                'raudra': 'Raudra (Anger)',

                'veera': 'Veera (Heroism)',

                'bhayanaka': 'Bhayanaka (Fear)',

                'bibhatsa': 'Bibhatsa (Disgust)',

                'adbhuta': 'Adbhuta (Wonder)',

                'shanta': 'Shanta (Peace)'

            };

            return rasaNames[rasa] || rasa;

        }

        

        function deleteNote(id) {

            const oldLength = notes.length;

            notes = notes.filter(note => note.id !== id);

            

            saveNotes();

            loadNotes();

            updateCounter();

        }

        

        function clearAllNotes() {

            if (notes.length === 0) {

                return;

            }

            

            notes.length = 0;

            saveNotes();

            

            const container = document.getElementById('notesContainer');

            container.innerHTML = `

                <div class="empty-state">

                    No notes yet! Go to "Add Note" tab to create your first note.

                </div>

            `;

            

            updateCounter();

        }

        

        function saveNotes() {

            localStorage.setItem('navrasaNotes', JSON.stringify(notes));

        }

        

        function updateCounter() {

            const count = notes.length;

            const counterText = `Total notes: ${count}`;

            

            const addTabCounter = document.getElementById('counter');

            if (addTabCounter) {

                addTabCounter.textContent = counterText;

            }

        }

        

        function escapeHtml(text) {

            const div = document.createElement('div');

            div.textContent = text;

            return div.innerHTML;

        }

        

        // Add note with Ctrl+Enter

        document.getElementById('noteInput').addEventListener('keydown', function(e) {

            if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') {

                addNote();

            }

        });

        

        // Initialize

        updateCounter();

    </script>
